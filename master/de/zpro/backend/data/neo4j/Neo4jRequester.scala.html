31   private def run(details: RequestObject, whereString: String): Future[List[DbResult]] =
32     buildRequest(details, whereString).flatMap(request =>
33       connection.executeRequest(request)
34         .map(records => parser.parseResult(records, details))
35     )
36 
37   private def buildRequest(details: RequestObject, whereString: String): Future[String] = {
38     val simplePropertiesRequest = complexGenerator.filterOutComplex(details)
39     matchGenerator.generateMatchString(simplePropertiesRequest) match {
40       case Success(matchString) =>
41         val propertyCalls = complexGenerator.generateComplex(details)
42         val returnString = returnGenerator.generateReturnString(details)
43         Future.successful(
44           s"""$matchString
45              |$whereString
46              |$propertyCalls
47              |$returnString""".stripMargin)
48       case Failure(e: UnknownRelationException) => Future.failed(ParsingException(e.getMessage, e))
49       case Failure(e) => Future.failed(e)
50     }
51   }
52 
53   private def runQuickSearch(request: String): Future[List[DbResult]] = {
54     val response = connection.executeRequest(request)
55     response.map(records =>
56       records.map(record => {
57         val result = record.get("node").asNode()
58         val node = DbResult(
59           dataType = result.labels().iterator().next(),
60           properties = Map(
61             "id" -> result.get("id"),
62             "fullName" -> result.get("fullName")
63           )
64         )
65         DbResult(
66           dataType = "SearchResult",
67           properties = Map("score" -> record.get("score")),
68           nodes = Map("node" -> List(node))
69         )
70       })
71     )
72   }
73 }
