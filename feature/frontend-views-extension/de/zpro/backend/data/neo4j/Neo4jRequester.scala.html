30   private def run(details: RequestObject, id: Option[String] = None): Future[List[DbResult]] =
31     buildRequest(details, id) match {
32       case Success(request) =>
33         connection.executeRequest(request)
34           .map(records => parser.parseResult(records, details))
35       case Failure(e) => Future.failed(e)
36     }
37 
38   private def buildRequest(details: RequestObject, id: Option[String]): Try[String] =
39     generator.generateReturn(details) match {
40       case Success(returnString) => Success(
41         s"""MATCH (${details.name}:${details.dataType})
42            |${id.map(id => s"WHERE ${details.name}.id = \"$id\"").getOrElse("")}
43            |$returnString""".stripMargin)
44       case Failure(e: UnknownRelationException) => Failure(ParsingException(e.getMessage, e))
45       case f: Failure[_] => f
46     }
47 
48   private def runQuickSearch(request: String): Future[List[DbResult]] = {
49     val response = connection.executeRequest(request)
50     response.map(records =>
51       records.map(record => {
52         val result = record.get("node").asNode()
53         val node = DbResult(
54           dataType = result.labels().iterator().next(),
55           properties = Map(
56             "id" -> result.get("id"),
57             "fullName" -> result.get("fullName")
58           )
59         )
60         DbResult(
61           dataType = "SearchResult",
62           properties = Map("score" -> record.get("score")),
63           nodes = Map("node" -> List(node))
64         )
65       })
66     )
67   }
