29   private def run(details: RequestObject, whereString: String): Future[List[DbResult]] =
30     buildRequest(details, whereString).flatMap(request =>
31       connection.executeRequest(request)
32         .map(records => parser.parseResult(records, details))
33     )
34 
35   private def buildRequest(details: RequestObject, whereString: String): Future[String] = {
36     matchGenerator.generateMatchString(details) match {
37       case Success(matchString) =>
38         val returnString = returnGenerator.generateReturnString(details)
39         Future.successful(s"$matchString $whereString $returnString")
40       case Failure(e: UnknownRelationException) => Future.failed(ParsingException(e.getMessage, e))
41       case Failure(e) => Future.failed(e)
42     }
43   }
44 
45   private def runQuickSearch(request: String): Future[List[DbResult]] = {
46     val response = connection.executeRequest(request)
47     response.map(records =>
48       records.map(record => {
49         val result = record.get("node").asNode()
50         val node = DbResult(
51           dataType = result.labels().iterator().next(),
52           properties = Map(
53             "id" -> result.get("id"),
54             "fullName" -> result.get("fullName")
55           )
56         )
57         DbResult(
58           dataType = "SearchResult",
59           properties = Map("score" -> record.get("score")),
60           nodes = Map("node" -> List(node))
61         )
62       })
63     )
64   }
